# Install Docker Desktop and standard tools
Write-Host "Initializing LabSync deployment on $env:COMPUTERNAME..."

# Define packages
$packages = @("Docker.DockerDesktop", "Microsoft.VisualStudioCode", "Git.Git")

foreach ($pkg in $packages) {
  Write-Host "Installing $pkg..."
  winget install -e --id $pkg --accept-package-agreements --accept-source-agreements
  if ($LASTEXITCODE -ne 0) {
    Write-Error "Failed to install $pkg"
  }
}

Write-Host "Deployment Complete."
